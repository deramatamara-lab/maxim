import { ds } from '@/constants/theme';
 
type OrbitControlsRef = React.ElementRef<typeof OrbitControls>;

  zoomToLocation: (coords: { lat: number; lon: number }) => void;

// Utility: Convert lat/lng to 3D vector on sphere
const latLngToVector3 = (lat: number, lng: number, radius: number): THREE.Vector3 => {
  const phi = (90 - lat) * (Math.PI / 180);
  const theta = (lng + 180) * (Math.PI / 180);
  const x = -(radius * Math.sin(phi) * Math.cos(theta));
  const z = radius * Math.sin(phi) * Math.sin(theta);
  const y = radius * Math.cos(phi);
  return new THREE.Vector3(x, y, z);
};
 
// Generate random network arcs for traffic visualization
const generateNetworkArcs = (count: number, radius: number) => {
  const arcs = [];
  for (let i = 0; i < count; i++) {
    const startLat = (Math.random() - 0.5) * 160;
    const startLng = (Math.random() - 0.5) * 360;
    const endLat = (Math.random() - 0.5) * 160;
    const endLng = (Math.random() - 0.5) * 360;
    
    if (Math.abs(startLat - endLat) < 10) continue;
    
    arcs.push({
      start: latLngToVector3(startLat, startLng, radius),
      end: latLngToVector3(endLat, endLng, radius),
      mid: latLngToVector3(
        (startLat + endLat) / 2,
        (startLng + endLng) / 2,
        radius * (1.2 + Math.random() * 0.3)
      ),
    });
  }
  return arcs;
};
 
// Atmosphere shader material
const AtmosphereShaderMaterial = {
  uniforms: {
    color: { value: new THREE.Color(ds.colors.primary) },
  },
  vertexShader: `
    varying vec3 vNormal;
    varying vec3 vViewPosition;
    void main() {
      vNormal = normalize(normalMatrix * normal);
      vec4 mvPosition = modelViewMatrix * vec4(position, 1.0);
      vViewPosition = -mvPosition.xyz;
      gl_Position = projectionMatrix * mvPosition;
    }
  `,
  fragmentShader: `
    uniform vec3 color;
    varying vec3 vNormal;
    varying vec3 vViewPosition;
    void main() {
      vec3 normal = normalize(vNormal);
      vec3 viewDir = normalize(vViewPosition);
      float intensity = pow(0.6 - dot(normal, viewDir), 2.5);
      gl_FragColor = vec4(color, 1.0) * intensity * 1.5;
    }
  `,
};
 
// Animated Network Arc Component
function NetworkArc({ 
  start, 
  end, 
  mid, 
  delay 
}: { 
  start: THREE.Vector3; 
  end: THREE.Vector3; 
  mid: THREE.Vector3; 
  delay: number;
}) {
  const [active, setActive] = useState(false);
  
  useEffect(() => {
    const timer = setTimeout(() => setActive(true), delay);
    return () => clearTimeout(timer);
  }, [delay]);
  
  if (!active) return null;
  
  return (
    <QuadraticBezierLine
      start={start}
      end={end}
      mid={mid}
      color={ds.colors.secondary}
      lineWidth={1}
      transparent
      opacity={0.4}
      dashed
      dashScale={50}
      dashSize={2}
    />
  );
}
 
// Holographic Earth Component
function HolographicEarth({ isZoomed = false }: { isZoomed?: boolean }) {
  const groupRef = useRef<THREE.Group>(null);
  const atmosphereRef = useRef<THREE.Mesh>(null);
  const arcs = useMemo(() => generateNetworkArcs(12, 2), []);
  
  useFrame((state, delta) => {
    // Pulse atmosphere
    if (atmosphereRef.current) {
      const material = atmosphereRef.current.material as THREE.ShaderMaterial;
      if (material.opacity !== undefined) {
        material.opacity = 0.5 + Math.sin(state.clock.elapsedTime) * 0.1;
      }
    }
    
    // Rotate when not zoomed
    if (groupRef.current && !isZoomed) {
      groupRef.current.rotation.y += delta * 0.02;
    }
  });
  
  return (
    <group ref={groupRef}>
      {/* 1. Black Glossy Core */}
      <Sphere args={[1.98, 64, 64]}>
        <meshStandardMaterial
          color="#000000"
          roughness={0.1}
          metalness={0.9}
        />
      </Sphere>
      
      {/* 2. Wireframe Grid */}
      <Sphere args={[2, 32, 32]}>
        <meshBasicMaterial
          color={ds.colors.primary}
          wireframe
          transparent
          opacity={0.08}
        />
      </Sphere>
      
      {/* 3. Dot Matrix Points */}
      <points>
        <sphereGeometry args={[2.01, 64, 64]} />
        <pointsMaterial
          size={0.015}
          color={ds.colors.primary}
          transparent
          opacity={0.2}
          sizeAttenuation
        />
      </points>
      
      {/* 4. Atmosphere Glow */}
      <mesh ref={atmosphereRef} scale={[1.2, 1.2, 1.2]}>
        <sphereGeometry args={[2, 64, 64]} />
        <shaderMaterial
          args={[AtmosphereShaderMaterial]}
          transparent
          side={THREE.BackSide}
          depthWrite={false}
          blending={THREE.AdditiveBlending}
        />
      </mesh>
      
      {/* 5. Network Traffic Arcs */}
      <group>
        {arcs.map((arc, i) => (
          <NetworkArc
            key={i}
            start={arc.start}
            end={arc.end}
            mid={arc.mid}
            delay={i * 500}
          />
        ))}
      </group>
    </group>
  );
}
 
// Location Marker Component
function LocationMarker({ lat, lng }: { lat: number; lng: number }) {
  const position = useMemo(() => latLngToVector3(lat, lng, 2.05), [lat, lng]);
  
  return (
    <group position={position}>
      {/* Pulsing Ring */}
      <mesh rotation={[Math.PI / 2, 0, 0]}>
        <ringGeometry args={[0.04, 0.05, 32]} />
        <meshBasicMaterial
          color={ds.colors.primary}
          side={THREE.DoubleSide}
          transparent
          opacity={0.8}
        />
      </mesh>
      {/* Vertical Line */}
      <mesh position={[0, 0, -0.1]}>
        <cylinderGeometry args={[0.005, 0.005, 0.5, 8]} />
        <meshBasicMaterial color={ds.colors.primary} transparent opacity={0.3} />
      </mesh>
      {/* Point Light */}
      <pointLight distance={0.5} intensity={2} color={ds.colors.primary} />
    </group>
  );
}
 
// Starfield component
function Starfield() {
  return <Stars radius={100} depth={50} count={3000} factor={4} saturation={0} fade speed={0.5} />;
}
 
// Camera Controller for zoom animations
function CameraController({ 
  target, 
  isZoomed 
}: { 
  target?: { lat: number; lng: number } | null;
  isZoomed: boolean;
}) {
  const { camera } = useThree();
  const controlsRef = useRef<OrbitControlsRef>(null);
  
  useFrame((_, delta) => {
    if (target && isZoomed && controlsRef.current) {
      const targetPos = latLngToVector3(target.lat, target.lng, 5);
      camera.position.lerp(targetPos, delta * 2);
      controlsRef.current.target.lerp(new THREE.Vector3(0, 0, 0), delta * 2);
    }
    controlsRef.current?.update();
  });
  
  return (
    <OrbitControls
      ref={controlsRef}
      enablePan={false}
      enableZoom={true}
      minDistance={3}
      maxDistance={10}
      autoRotate={!isZoomed}
      autoRotateSpeed={0.5}
      dampingFactor={0.05}
      enableDamping={true}
      rotateSpeed={0.5}
    />
  );
}
 
export interface GlobeProps {
  height?: number | string;
  targetCoordinates?: { lat: number; lng: number } | null;
  isZoomed?: boolean;
  onZoomComplete?: () => void;
}
 
export const Globe = React.forwardRef<GlobeHandle, GlobeProps>(
  ({ height = 300, targetCoordinates = null, isZoomed = false, onZoomComplete }, ref) => {
    const [internalTarget, setInternalTarget] = useState<{ lat: number; lng: number } | null>(null);
    const [internalZoomed, setInternalZoomed] = useState(false);
    
    // Use props or internal state
    const activeTarget = targetCoordinates ?? internalTarget;
    const activeZoomed = isZoomed || internalZoomed;
 
    React.useImperativeHandle(ref, () => ({
      zoomToLocation: (coords: { lat: number; lon: number }) => {
        setInternalTarget({ lat: coords.lat, lng: coords.lon });
        setInternalZoomed(true);
        
        setTimeout(() => {
          onZoomComplete?.();
        }, 1500);
      },
    }));
 
    return (
      <div style={{ height, width: '100%', position: 'relative' }}>
        <Canvas
          camera={{ position: [0, 0, 6], fov: 45 }}
          dpr={[1, 2]}
          style={{
            borderRadius: typeof ds.radius.lg === 'number' ? ds.radius.lg : 24,
            background: '#000',
          }}
        >
          <fog attach="fog" args={['#000', 5, 15]} />
          <ambientLight intensity={0.2} />
          <pointLight position={[10, 10, 10]} intensity={1.5} color="#4444ff" />
          <pointLight position={[-10, -5, -10]} intensity={0.5} color={ds.colors.primary} />
          
          <Starfield />
          <HolographicEarth isZoomed={activeZoomed} />
          
          {activeTarget && (
            <LocationMarker lat={activeTarget.lat} lng={activeTarget.lng} />
          )}
          
          <CameraController target={activeTarget} isZoomed={activeZoomed} />
        </Canvas>
        
        {/* Vignette Overlay */}
        <div
          style={{
            position: 'absolute',
            inset: 0,
            background: 'radial-gradient(circle at center, transparent 0%, #050607 120%)',
            pointerEvents: 'none',
            borderRadius: typeof ds.radius.lg === 'number' ? ds.radius.lg : 24,
          }}
        />
      </div>
    );
  }
);
